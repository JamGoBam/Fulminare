src/app/App.tsx
import { useState } from 'react';
import { Sidebar } from './components/Sidebar';
import { Header } from './components/Header';
import { KPICards } from './components/KPICards';
import { FilterBar } from './components/FilterBar';
import { ActionQueue } from './components/ActionQueue';
import { RecommendationPanel } from './components/RecommendationPanel';
import { InventoryScreen } from './components/InventoryScreen';
import { AnalyticsScreen } from './components/AnalyticsScreen';
import { ReportsScreen } from './components/ReportsScreen';
import { actionQueueData } from './data/actionQueueData';

export default function App() {
  const [selectedItemId, setSelectedItemId] = useState<string | null>(null);
  const [currentView, setCurrentView] = useState<string>('dashboard');

  const selectedItem = actionQueueData.find(item => item.id === selectedItemId) || null;

  const renderMainContent = () => {
    switch (currentView) {
      case 'inventory':
        return <InventoryScreen />;
      case 'analytics':
        return <AnalyticsScreen />;
      case 'reports':
        return <ReportsScreen />;
      case 'settings':
        return (
          <div className="p-8">
            <h1 className="text-2xl font-semibold text-slate-900 mb-2">Settings</h1>
            <p className="text-slate-600">Configuration and preferences</p>
          </div>
        );
      default:
        return (
          <div className="p-8">
            <div className="mb-6">
              <h1 className="text-2xl font-semibold text-slate-900 mb-2">Operations Dashboard</h1>
              <p className="text-slate-600">Manage inventory imbalances across 3 DCs • Prevent chargebacks and maintain OTIF performance</p>
            </div>

            <div className="mb-6">
              <KPICards />
            </div>

            <FilterBar />

            <div className="grid grid-cols-3 gap-6">
              <div className="col-span-2">
                <ActionQueue
                  items={actionQueueData}
                  selectedId={selectedItemId}
                  onSelectItem={setSelectedItemId}
                />
              </div>

              <div className="col-span-1">
                <RecommendationPanel item={selectedItem} />
              </div>
            </div>
          </div>
        );
    }
  };

  return (
    <div className="flex h-screen bg-slate-50">
      <Sidebar currentView={currentView} onNavigate={setCurrentView} />

      <div className="flex-1 flex flex-col overflow-hidden">
        <Header />

        <main className="flex-1 overflow-auto">
          {renderMainContent()}
        </main>
      </div>
    </div>
  );
}

src/app/components/ActionQueue.tsx
import { Clock, TrendingUp, AlertCircle, CheckCircle } from 'lucide-react';
import { ActionItem } from '../data/actionQueueData';

interface ActionQueueProps {
  items: ActionItem[];
  selectedId: string | null;
  onSelectItem: (id: string) => void;
}

export function ActionQueue({ items, selectedId, onSelectItem }: ActionQueueProps) {
  const getRiskBadgeStyles = (risk: string) => {
    switch (risk) {
      case 'High':
        return 'bg-red-100 text-red-700 border-red-200';
      case 'Medium':
        return 'bg-amber-100 text-amber-700 border-amber-200';
      case 'Low':
        return 'bg-green-100 text-green-700 border-green-200';
      default:
        return 'bg-slate-100 text-slate-700 border-slate-200';
    }
  };

  const getRecommendationBadgeStyles = (rec: string) => {
    switch (rec) {
      case 'Transfer Now':
        return 'bg-blue-600 text-white';
      case 'Wait':
        return 'bg-slate-600 text-white';
      case 'Escalate':
        return 'bg-purple-600 text-white';
      default:
        return 'bg-slate-600 text-white';
    }
  };

  const getUrgencyIndicator = (days: number) => {
    if (days <= 1) return { color: 'text-red-600', icon: AlertCircle, bg: 'bg-red-50' };
    if (days <= 3) return { color: 'text-amber-600', icon: Clock, bg: 'bg-amber-50' };
    return { color: 'text-green-600', icon: CheckCircle, bg: 'bg-green-50' };
  };

  return (
    <div className="bg-white rounded-xl border border-slate-200 shadow-sm">
      <div className="border-b border-slate-200 p-6">
        <h2 className="text-xl font-semibold text-slate-900">Action Queue</h2>
        <p className="text-sm text-slate-500 mt-1">Prioritized inventory decisions requiring immediate attention</p>
      </div>

      <div className="overflow-auto max-h-[calc(100vh-280px)]">
        {items.map((item, index) => {
          const urgency = getUrgencyIndicator(item.daysUntilStockout);
          const UrgencyIcon = urgency.icon;
          const isSelected = selectedId === item.id;
          const isTopPriority = index < 3 && item.riskLevel === 'High';

          return (
            <button
              key={item.id}
              onClick={() => onSelectItem(item.id)}
              className={`w-full text-left p-5 border-b border-slate-100 hover:bg-slate-50 transition-colors relative ${
                isSelected ? 'bg-blue-50 border-l-4 border-l-blue-600' : 'border-l-4 border-l-transparent'
              } ${isTopPriority ? 'bg-red-50/30 border-l-4 border-l-red-500' : ''}`}
            >
              {isTopPriority && (
                <div className="absolute top-2 right-2">
                  <span className="px-2 py-1 bg-red-600 text-white text-xs font-semibold rounded">
                    URGENT
                  </span>
                </div>
              )}
              <div className="flex items-start gap-4">
                <div className={`${urgency.bg} p-3 rounded-lg flex-shrink-0`}>
                  <UrgencyIcon className={`w-5 h-5 ${urgency.color}`} />
                </div>

                <div className="flex-1 min-w-0">
                  <div className="flex items-start justify-between gap-4 mb-2">
                    <div>
                      <div className="font-semibold text-slate-900">{item.itemName}</div>
                      <div className="text-sm text-slate-500">{item.sku} • {item.category}</div>
                    </div>
                    <div className="flex items-center gap-2 flex-shrink-0">
                      <span className={`px-3 py-1 rounded-full text-xs font-medium ${getRecommendationBadgeStyles(item.recommendation)}`}>
                        {item.recommendation}
                      </span>
                      <span className={`px-3 py-1 rounded-full text-xs font-medium border ${getRiskBadgeStyles(item.riskLevel)}`}>
                        {item.riskLevel} Risk
                      </span>
                    </div>
                  </div>

                  <div className="grid grid-cols-5 gap-4 mb-3">
                    <div>
                      <div className="text-xs text-slate-500 mb-1">At-Risk DC</div>
                      <div className="text-sm font-medium text-slate-900">{item.atRiskDC}</div>
                    </div>
                    <div>
                      <div className="text-xs text-slate-500 mb-1">Company-Wide Inv.</div>
                      <div className="text-sm font-medium text-slate-900">{item.companyWideInventory}</div>
                    </div>
                    <div>
                      <div className="text-xs text-slate-500 mb-1">Days to Stockout</div>
                      <div className={`text-sm font-semibold ${urgency.color}`}>{item.daysUntilStockout} days</div>
                    </div>
                    <div>
                      <div className="text-xs text-slate-500 mb-1">Potential Penalty</div>
                      <div className="text-sm font-semibold text-slate-900">{item.potentialPenalty}</div>
                    </div>
                    <div>
                      <div className="text-xs text-slate-500 mb-1">Confidence</div>
                      <div className="flex items-center gap-2">
                        <div className="flex-1 bg-slate-200 rounded-full h-1.5">
                          <div
                            className="bg-blue-600 h-1.5 rounded-full"
                            style={{ width: `${item.confidence}%` }}
                          ></div>
                        </div>
                        <span className="text-sm font-medium text-slate-900">{item.confidence}%</span>
                      </div>
                    </div>
                  </div>

                  <div className="flex items-center justify-between">
                    <div className="flex items-center gap-2 text-sm text-slate-600">
                      <TrendingUp className="w-4 h-4" />
                      <span>{item.reason}</span>
                    </div>
                    <div className="text-xs text-slate-400">{item.updatedAt}</div>
                  </div>
                </div>
              </div>
            </button>
          );
        })}
      </div>
    </div>
  );
}
